Callbacks

Mistakes that many beginners make


for (var i = 0; i < 5; i++) {
    setTimeout(function () {
      console.log(i);
    }, i * 1000);
}
// vs
for (var i = 0; i < 5; i++) {
    function print (x) {
      setTimeout(function () {
        console.log(x);
      }, x * 1000);
    }
    print(i);
}
